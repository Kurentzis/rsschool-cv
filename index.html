                    def square_digits(num):  
                    x = [int(a) for a in str(num)]
                    squared_nums = []

                        squared_num = str(squared_num)
                        squared_nums.append(squared_num)
                    separator = ''
                    result = separator.join(squared_nums)
                    result = int(result)
                    return (result)
